CREATE TABLE IF NOT EXISTS RESTAURANT (
   RESTAURANT_ID INT,
   RESTAURANT_NAME VARCHAR(255) ,
   VEGAN VARCHAR(100),
   AGGREGATE_RATING DECIMAL,
   DELIVERABLE VARCHAR(100),
    PRIMARY KEY (RESTAURANT_ID)
);  

create table if not exists CUISINE(
CUISINE_ID INT,
CUISINE_NAME VARCHAR(100),
PRIMARY KEY(CUISINE_ID)
);

CREATE TABLE IF NOT EXISTS RATING_RESTAURANT (
   RESTAURANT_ID INT,
   AVG_RATING_COLOUR VARCHAR(255) ,
   RATING_TEXT VARCHAR(255),
   PRICE_RANGE_RATING INT,
   HOSPITALITY_RATING DECIMAL,
   AVG_COST_PER_TWO INT,
    FOREIGN KEY (RESTAURANT_ID) REFERENCES RESTAURANT (RESTAURANT_ID)
);

create table if not exists RESTAURANT_TIMINGS(
RESTAURANT_ID INT,
OPENING_TIME VARCHAR(100),
CLOSING_TIME VARCHAR(100),
BREAK_TIME VARCHAR(100),
CLOSED_ON_DAYS VARCHAR(100),
EXTRA_INFO VARCHAR(100),
FOREIGN KEY (RESTAURANT_ID) REFERENCES RESTAURANT (RESTAURANT_ID)
);

Create table if not exists FOOD_ITEMS(
CUISINE_ID INT,
RESTAURANT_ID INT,
FOOD_ITEMS_TOP VARCHAR(255),
PRICE INT,
FOREIGN KEY (RESTAURANT_ID) REFERENCES RESTAURANT (RESTAURANT_ID),
FOREIGN KEY (CUISINE_ID) REFERENCES CUISINE (CUISINE_ID)
);


Create table IF NOT EXISTS CITY_STATE(
CITY_ID INT,
CITY varchar(100),
STATE varchar(100),
PRIMARY KEY(CITY_ID)

);

Create table IF NOT EXISTS PIN_CODE(
PINCODE INT,
LOCALITY varchar(100),
CITY_ID INT,
PRIMARY KEY(PINCODE),
FOREIGN KEY (CITY_ID) REFERENCES CITY_STATE(CITY_ID)
);



Create table IF NOT EXISTS ADDRESS(
RESTAURANT_ID INT,
BUILDING_NUMBER varchar(100),
STREET_NAME varchar(100),
PINCODE int,
FOREIGN KEY (RESTAURANT_ID) REFERENCES RESTAURANT (RESTAURANT_ID),
FOREIGN KEY (PINCODE) REFERENCES PIN_CODE (PINCODE)
);
